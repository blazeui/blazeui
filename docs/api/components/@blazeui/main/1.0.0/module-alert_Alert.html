Example
{{#Alert}}
    {{> Icon name="solid-bolt" }}
    {{#AlertTitle}}Info{{/AlertTitle}}
    {{#AlertDescription}}
        This is a default alert
    {{/AlertDescription}}
{{/Alert}}

{{#Alert variant="primary" class="mt-5"}}
    {{> Icon name="solid-bolt" }}
    {{#AlertTitle}}Info{{/AlertTitle}}
    {{#AlertDescription}}
        This is a primary alert
    {{/AlertDescription}}
{{/Alert}}

{{#Alert variant="destructive" class="mt-5"}}
    {{> Icon name="solid-skull-crossbones"}}
    {{#AlertTitle}}Caution!{{/AlertTitle}}
    {{#AlertDescription}}
        This is highly important content.
        {{> Icon name="solid-bolt"}}
    {{/AlertDescription}}
{{/Alert}}
